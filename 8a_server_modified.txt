"""
Flask application for emotion detection.

This module exposes endpoints to analyze text and return
emotion scores along with the dominant emotion.
"""

from flask import Flask, render_template, request
from EmotionDetection.emotion_detection import emotion_detector

# Instantiate the Flask app
app = Flask(__name__)


@app.route("/emotionDetector")
def sent_analyzer():
    """
    Analyze user-provided text and return detected emotions.

    Returns:
        str: Formatted string with emotion scores and dominant emotion.
    """
    text_to_analyze = request.args.get("textToAnalyze", default="", type=str)

    # Handle empty input early
    if not text_to_analyze.strip():
        return "Invalid text! Please try again!"

    try:
        response = emotion_detector(text_to_analyze)
    except Exception:  # pylint: disable=broad-except
        return "Error processing the request. Please try again!"

    dominant_emotion = response.get("dominant_emotion")

    if dominant_emotion is None:
        return "Invalid text! Please try again!"

    return (
        "For the given statement, the system response is "
        f"'anger': {response.get('anger')}, "
        f"'disgust': {response.get('disgust')}, "
        f"'fear': {response.get('fear')}, "
        f"'joy': {response.get('joy')} and "
        f"'sadness': {response.get('sadness')}. "
        f"The dominant emotion is {dominant_emotion}."
    )


@app.route("/")
def render_index_page():
    """
    Render the main application page.

    Returns:
        str: HTML content of the index page.
    """
    return render_template("index.html")


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
